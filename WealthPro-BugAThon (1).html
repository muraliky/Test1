import { spawnSync } from "child_process"; import path from "path"; import fs from "fs"; const BAT_FILE_PATH = path.resolve(__dirname, "../../captureScreenshots.bat"); const BASE_OUTPUT_DIR = path.resolve(__dirname, "../../test-results/screenshots"); function generateRunTimestamp(): string { const now = new Date(); const pad = (n: number) => String(n).padStart(2, "0"); return `${now.getFullYear()}-${pad(now.getMonth() + 1)}-${pad(now.getDate())}` + `_${pad(now.getHours())}-${pad(now.getMinutes())}-${pad(now.getSeconds())}`; } const RUN_TIMESTAMP = generateRunTimestamp(); const RUN_OUTPUT_DIR = path.join(BASE_OUTPUT_DIR, RUN_TIMESTAMP); export function captureMainframeScreenshots(screens: string, accountNumber: string): void { const trimmedScreens = screens?.trim(); const trimmedAccount = accountNumber?.trim(); if (!trimmedScreens) { console.log("[Screenshot] Screens column is empty — skipping capture."); return; } if (!trimmedAccount) { throw new Error("[Screenshot] Account Number is required but was empty."); } fs.mkdirSync(RUN_OUTPUT_DIR, { recursive: true }); console.log(`[Screenshot] Triggering BAT for screens: ${trimmedScreens}, account: ${trimmedAccount}`); console.log(`[Screenshot] Output directory: ${RUN_OUTPUT_DIR}`); // IMPORTANT: when the BAT path itself contains spaces, cmd.exe /c needs // the ENTIRE command line wrapped as one quoted string, not separate // array elements — otherwise cmd.exe mis-parses the quoted path. const commandLine = `"${BAT_FILE_PATH}" "${trimmedScreens}" "${trimmedAccount}" "${RUN_OUTPUT_DIR}"`; const result = spawnSync("cmd.exe", ["/c", commandLine], { stdio: "inherit", encoding: "utf-8", shell: false, }); if (result.error) { throw new Error(`[Screenshot] Failed to spawn BAT process: ${result.error.message}`); } if (result.status !== 0) { throw new Error( `[Screenshot] BAT process exited with code ${result.status}. Screens: ${trimmedScreens}, Account: ${trimmedAccount}` ); } console.log(`[Screenshot] Mainframe capture completed for: ${trimmedScreens}`); }
