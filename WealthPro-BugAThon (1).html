package com.yourpackage.excel; import org.apache.poi.ss.usermodel.*; import org.apache.poi.xssf.usermodel.XSSFWorkbook; import org.apache.poi.util.Units; import javax.imageio.ImageIO; import java.awt.image.BufferedImage; import java.io.File; import java.io.FileOutputStream; import java.io.IOException; import java.nio.file.Files; public class ExcelScreenshotWriter { private final Workbook workbook; private final Sheet sheet; private int currentRow = 0; private final String outputExcelPath; // Target display size for every screenshot, in pixels. // Tune these two values to whatever fits your report layout. private static final int TARGET_WIDTH_PX = 480; private static final int TARGET_HEIGHT_PX = 320; public ExcelScreenshotWriter(String outputExcelPath, String sheetName) { this.outputExcelPath = outputExcelPath; this.workbook = new XSSFWorkbook(); this.sheet = workbook.createSheet(sheetName); Row header = sheet.createRow(currentRow++); header.createCell(0).setCellValue("Screen Name"); header.createCell(1).setCellValue("Screenshot"); // Fix column B's width once, so every image lines up consistently. // Column width units are roughly 1/256 of a character width. sheet.setColumnWidth(1, widthPxToColumnUnits(TARGET_WIDTH_PX)); } public void addScreenshotRow(String screenName, String imagePath) throws IOException { Row row = sheet.createRow(currentRow); row.createCell(0).setCellValue(screenName); byte[] imageBytes = Files.readAllBytes(new File(imagePath).toPath()); int pictureIdx = workbook.addPicture(imageBytes, Workbook.PICTURE_TYPE_PNG); // Read actual captured dimensions so we scale proportionally, // not stretch/distort the screenshot. BufferedImage original = ImageIO.read(new File(imagePath)); double scale = computeScaleToFit(original.getWidth(), original.getHeight()); int scaledWidthPx = (int) (original.getWidth() * scale); int scaledHeightPx = (int) (original.getHeight() * scale); CreationHelper helper = workbook.getCreationHelper(); Drawing drawing = sheet.createDrawingPatriarch(); ClientAnchor anchor = helper.createClientAnchor(); // MOVE_DONT_RESIZE: anchor point fixed, image size controlled // purely by our explicit scale call below, not by cell resizing. anchor.setAnchorType(ClientAnchor.AnchorType.MOVE_DONT_RESIZE); anchor.setCol1(1); anchor.setRow1(currentRow); Picture picture = drawing.createPicture(anchor, pictureIdx); picture.resize(scale); // scales relative to the image's native EMU size // Row height must match the scaled image height, converted to points. float rowHeightPoints = pixelsToPoints(scaledHeightPx); row.setHeightInPoints(rowHeightPoints); currentRow++; } private double computeScaleToFit(int originalWidthPx, int originalHeightPx) { double widthScale = (double) TARGET_WIDTH_PX / originalWidthPx; double heightScale = (double) TARGET_HEIGHT_PX / originalHeightPx; // use the smaller scale so the image fits within BOTH bounds, // preserving aspect ratio instead of stretching return Math.min(widthScale, heightScale); } private float pixelsToPoints(int pixels) { // 1 point = 1.333 pixels at 96 DPI (standard screen resolution) return pixels / 1.333f; } private int widthPxToColumnUnits(int pixels) { // Rough conversion: 1 character unit ≈ 7 pixels at default font/zoom. // POI column width is in 1/256ths of a character width. int charWidthUnits = (pixels / 7); return charWidthUnits * 256; } public void save() throws IOException { try (FileOutputStream fos = new FileOutputStream(outputExcelPath)) { workbook.write(fos); } workbook.close(); System.out.println("[Java] Excel file saved: " + outputExcelPath); } }
