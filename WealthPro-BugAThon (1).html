package com.yourpackage.word; import org.apache.poi.util.Units; import org.apache.poi.xwpf.usermodel.*; import javax.imageio.ImageIO; import java.awt.image.BufferedImage; import java.io.File; import java.io.FileInputStream; import java.io.FileOutputStream; import java.io.IOException; public class WordScreenshotWriter { private final XWPFDocument document; private final String outputDocxPath; // Target display size for every screenshot, in pixels. // Tune to whatever fits your report layout / page width. private static final int TARGET_WIDTH_PX = 500; private static final int TARGET_HEIGHT_PX = 340; public WordScreenshotWriter(String outputDocxPath, String reportTitle) { this.outputDocxPath = outputDocxPath; this.document = new XWPFDocument(); XWPFParagraph titlePara = document.createParagraph(); titlePara.setAlignment(ParagraphAlignment.CENTER); XWPFRun titleRun = titlePara.createRun(); titleRun.setText(reportTitle); titleRun.setBold(true); titleRun.setFontSize(18); } /** * Adds a heading with the screen name, followed by the embedded * screenshot scaled to fit within the target box. */ public void addScreenshotSection(String screenName, String imagePath) throws IOException { // Section heading XWPFParagraph heading = document.createParagraph(); heading.setSpacingBefore(300); XWPFRun headingRun = heading.createRun(); headingRun.setText(screenName); headingRun.setBold(true); headingRun.setFontSize(14); // Read real dimensions so we scale proportionally, not distorted BufferedImage original = ImageIO.read(new File(imagePath)); double scale = computeScaleToFit(original.getWidth(), original.getHeight()); int scaledWidthPx = (int) (original.getWidth() * scale); int scaledHeightPx = (int) (original.getHeight() * scale); // Image paragraph XWPFParagraph imgPara = document.createParagraph(); imgPara.setAlignment(ParagraphAlignment.CENTER); XWPFRun imgRun = imgPara.createRun(); try (FileInputStream imgStream = new FileInputStream(imagePath)) { imgRun.addPicture( imgStream, XWPFDocument.PICTURE_TYPE_PNG, imagePath, Units.toEMU(scaledWidthPx), Units.toEMU(scaledHeightPx) ); } catch (org.apache.poi.openxml4j.exceptions.InvalidFormatException e) { throw new IOException("Invalid image format for " + imagePath, e); } } private double computeScaleToFit(int originalWidthPx, int originalHeightPx) { double widthScale = (double) TARGET_WIDTH_PX / originalWidthPx; double heightScale = (double) TARGET_HEIGHT_PX / originalHeightPx; return Math.min(widthScale, heightScale); // preserve aspect ratio } public void save() throws IOException { try (FileOutputStream fos = new FileOutputStream(outputDocxPath)) { document.write(fos); } document.close(); System.out.println("[Java] Word document saved: " + outputDocxPath); } } package com.yourpackage.runner; import com.yourpackage.capture.ScreenshotCapture; import com.yourpackage.word.WordScreenshotWriter; import com.yourpackage.mainframe.MainframeNavigator; public class ScreenshotOrchestrator { private final String outputDir; public ScreenshotOrchestrator(String outputDir) { this.outputDir = outputDir; } public boolean runForScreens(String[] screens) { MainframeNavigator navigator = new MainframeNavigator(); try { ScreenshotCapture capture = new ScreenshotCapture(outputDir); String docxFileName = "MainframeScreenshots_" + timestampFromOutputDir() + ".docx"; WordScreenshotWriter wordWriter = new WordScreenshotWriter(outputDir + "/" + docxFileName, "Mainframe Screenshot Report"); for (String screenName : screens) { navigator.navigateTo(screenName); String imagePath = capture.captureScreen(screenName); wordWriter.addScreenshotSection(screenName, imagePath); } wordWriter.save(); return true; } catch (Exception e) { System.err.println("[Java] ERROR: " + e.getMessage()); e.printStackTrace(); return false; } } private String timestampFromOutputDir() { String[] parts = outputDir.replace("\\", "/").split("/"); return parts[parts.length - 1]; } }
